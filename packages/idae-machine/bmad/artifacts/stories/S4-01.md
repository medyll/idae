## User Story

## Technical Requirements

### Router Configuration

```typescript
import { createRouter } from '@medyll/idae-router';

const router = createRouter({
  mode: 'history',
  base: '/app',
  outlet: '#app',
  routes: [
    { path: '/:table', action: renderList, metadata: { permission: 'L' } },
    { path: '/:table/new', action: renderCreate, metadata: { permission: 'C' } },
    { path: '/:table/:id', action: renderDetail, metadata: { permission: 'R' } },
    { path: '/:table/:id/edit', action: renderEdit, metadata: { permission: 'U' } }
  ]
});

// Permission guard
router.beforeEach((to, from, next) => {
  const requiredPerm = to.metadata?.permission;
  if (!hasPermission(requiredPerm, to.params.table)) {
    next('/403');
  } else {
    next();
  }
});
```
