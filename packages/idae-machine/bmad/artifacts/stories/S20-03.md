## API

```ts
// server/src/services/MailService.ts
export interface SendMailInput {
	to:       string | string[];
	cc?:      string | string[];
	bcc?:     string | string[];
	subject:  string;
	text?:    string;
	html?:    string;
	from?:    string;       // default: config.mail.from
	replyTo?: string;
	attachments?: Array<{
		filename: string;
		content:  Buffer | NodeJS.ReadableStream;
		contentType?: string;
	}>;
}

export interface SendTemplateInput {
	to:       string | string[];
	template: string;       // nom de fichier sans .md (ex: 'welcome', 'password-reset')
	vars:     Record<string, unknown>;
	subject?: string;       // override; sinon front-matter du template
	from?:    string;
	replyTo?: string;
}

export async function sendMail(input: SendMailInput): Promise<{ messageId: string } | { skipped: true; reason: string }>;
export async function sendTemplate(input: SendTemplateInput): Promise<{ messageId: string } | { skipped: true; reason: string }>;
export function renderTemplate(name: string, vars: Record<string, unknown>): { subject: string; text: string; html: string };
```

## Template format

Markdown avec front-matter YAML pour subject :

```
---
subject: "Bienvenue {{name}}"
---
# Bienvenue {{name}}

Ton compte a été créé. Login : **{{login}}**.

[Active ton compte]({{activationUrl}})
```

- Front-matter parsé via regex simple (`/^---\n(.+?)\n---\n/s` + ligne `subject:`) — pas besoin de `gray-matter`.
- Interpolation `{{var}}` simple via `String.prototype.replaceAll` — pas de Handlebars.
- HTML = markdown rendu via `marked` (lightweight).
- Text = markdown brut avec `{{vars}}` interpolés.

```ts
function renderTemplate(name: string, vars: Record<string, unknown>) {
	const file = path.join(config.mail.templatesDir, `${name}.md`);
	const raw  = fs.readFileSync(file, 'utf8');
	const fm   = /^---\n([\s\S]+?)\n---\n([\s\S]*)$/.exec(raw);
	const meta: Record<string, string> = {};
	let body = raw;
	if (fm) {
		for (const line of fm[1].split('\n')) {
			const m = /^(\w+):\s*"?(.*?)"?$/.exec(line.trim());
			if (m) meta[m[1]] = m[2];
		}
		body = fm[2];
	}
	const interp = (s: string) => s.replace(/\{\{(\w+)\}\}/g, (_, k) => String(vars[k] ?? ''));
	const text = interp(body);
	const html = marked.parse(text);
	const subject = interp(meta.subject ?? `(no subject)`);
	return { subject, text, html };
}
```

## Transport

```ts
import nodemailer from 'nodemailer';

let transporter: nodemailer.Transporter | null = null;

function getTransporter() {
	if (!config.mail.enabled) return null;
	if (transporter) return transporter;
	transporter = nodemailer.createTransport({
		host: config.mail.smtp.host,
		port: config.mail.smtp.port,
		secure: config.mail.smtp.secure,
		auth: config.mail.smtp.user
			? { user: config.mail.smtp.user, pass: config.mail.smtp.pass }
			: undefined,
	});
	return transporter;
}

export async function sendMail(input: SendMailInput) {
	const t = getTransporter();
	if (!t) {
		logger.info('[MailService disabled] would send:', input.subject, '→', input.to);
		return { skipped: true, reason: 'mail.enabled=false' };
	}
	const info = await t.sendMail({ ...input, from: input.from ?? config.mail.from });
	return { messageId: info.messageId };
}
```

## Retry simple

Pas de queue durable ce sprint. Wrapper avec retry mémoire :

```ts
async function withRetry<T>(fn: () => Promise<T>, attempts = 3, baseMs = 1000): Promise<T> {
	let lastErr: unknown;
	for (let i = 0; i < attempts; i++) {
		try { return await fn(); }
		catch (err) {
			lastErr = err;
			if (i < attempts - 1) await new Promise(r => setTimeout(r, baseMs * 2 ** i));
		}
	}
	throw lastErr;
}
```

## Acceptance criteria

1. `sendMail` avec `enabled=false` → retourne `{ skipped: true, ... }` + log
2. `sendMail` avec mock transport → `transporter.sendMail` appelé avec bonnes options
3. `renderTemplate('welcome', { name: 'Med' })` → subject + text + html interpolés
4. Front-matter absent → subject par défaut
5. Variable manquante `{{xxx}}` → remplacée par `''`
6. Retry : transport throw 2× puis succès → message envoyé
7. Retry : transport throw 3× → erreur propagée
8. Attachement Buffer → passé tel quel au transport
9. Endpoint `POST /api/mail/send` non-admin → 403
10. Suite no regression
