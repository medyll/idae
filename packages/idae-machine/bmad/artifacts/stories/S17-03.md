### Tests à écrire

```ts
describe('SchemeValidator', () => {
  beforeEach(() => invalidateSchemeCache());

  it('builds FieldRule map from appscheme', async () => {
    const rules = await getSchemeRules('vehicule');
    expect(rules.nomVehicule).toEqual({ required: true, type: 'text' });
    expect(rules.prixJour).toEqual({ required: true, type: 'number' });
    expect(rules.idVehicule).toEqual({ required: false }); // id skipped
  });

  it('caches result on second call', async () => {
    await getSchemeRules('vehicule');
    await getSchemeRules('vehicule');
    // getConn called once only (from first call)
    const { getConn } = await import('../middleware/dbRouter.js');
    expect(vi.mocked(getConn)).toHaveBeenCalledTimes(1);
  });

  it('returns {} for unknown collection gracefully', async () => {
    // Override mock to return empty
    const rules = await getSchemeRules('nonexistent');
    expect(rules).toEqual({});
  });

  it('validates required field missing → invalid', async () => {
    const result = await validateAgainstScheme('vehicule', { prixJour: 50 });
    expect(result.valid).toBe(false);
    expect(result.errors.nomVehicule).toBe('Ce champ est obligatoire');
  });

  it('validates valid record → valid', async () => {
    const result = await validateAgainstScheme('vehicule', { nomVehicule: 'Peugeot 308', prixJour: 50 });
    expect(result.valid).toBe(true);
  });

  it('validates type violation → invalid', async () => {
    const result = await validateAgainstScheme('vehicule', { nomVehicule: 'Peugeot', prixJour: 'pas-un-nombre' });
    expect(result.valid).toBe(false);
    expect(result.errors.prixJour).toBe('Doit être un nombre');
  });

  it('skips validation for collection with no schema', async () => {
    const result = await validateAgainstScheme('ghost_collection', {});
    expect(result.valid).toBe(true);
  });

  it('invalidateSchemeCache() clears all', async () => {
    await getSchemeRules('vehicule');
    invalidateSchemeCache();
    // Next call re-fetches
    const { getConn } = await import('../middleware/dbRouter.js');
    await getSchemeRules('vehicule');
    expect(vi.mocked(getConn)).toHaveBeenCalledTimes(2);
  });
});
```
