# S20-01 — Global hooks registry (`pre:*` / `post:*` unifié)

**Sprint:** 20
**Effort:** M
**Priority:** high
**Dependencies:** — (refactor sur l'existant `domainActions` + `AuditService` + `broadcastToTable`)

## Context

Legacy `actions.php` exposait `actions_pre.php` et `actions_post.php` — points d'extension globaux exécutés avant/après chaque CRUD. Mappé dans `LEGACY-ACTIONS-MIGRATION.md` section 4 sur `idae-db registerEvents()`.

État actuel serveur :
- `domainActions` (`server/src/models/domainActions.ts`) couvre les hooks **par collection** : `validate`, `afterCreate`, `afterUpdate`, `beforeDelete`.
- `routes/data.ts` câble en dur les concerns transverses : `logAudit(...)`, `broadcastToTable(...)`, validation, timestamps (Mongoose `timestamps:true`).
- Aucun moyen d'enregistrer un hook **global** sans toucher `routes/data.ts` (ex. plugin qui veut tracer toutes les écritures dans Slack, ou compteur métrique).

Goal : extraire un `HooksRegistry` qui factorise les hooks per-collection (`domainActions`) ET les hooks globaux. `routes/data.ts` n'invoque plus que `hooks.dispatch('pre:create', ctx)` / `hooks.dispatch('post:create', ctx)`. Audit + broadcast deviennent des hooks built-in.

## 1. Types

```ts
// server/src/hooks/types.ts
export type HookEvent =
	| 'pre:list'   | 'post:list'
	| 'pre:read'   | 'post:read'
	| 'pre:create' | 'post:create'
	| 'pre:update' | 'post:update'
	| 'pre:delete' | 'post:delete'
	| 'pre:restore'| 'post:restore'
	| 'error:create' | 'error:update' | 'error:delete';

export interface HookContext<T = unknown> {
	event:      HookEvent;
	collection: string;
	recordId?:  string;
	data?:      T;                       // body (pre) ou record (post)
	prev?:      T;                       // état avant update (post:update only)
	user?:      { userId?: string; login?: string };
	req?:       { ipAddress?: string; userAgent?: string };
	error?:     unknown;                 // error:* events
}

export type HookHandler = (ctx: HookContext) => void | Promise<void>;

export interface HookOptions {
	/** lower = earlier; default 100. Built-ins use 10–90 */
	priority?:   number;
	/** restrict to specific collection (default: all) */
	collection?: string;
	/** if true, exceptions in this hook abort the operation (default: false = fire-and-forget) */
	blocking?:   boolean;
}
```

## 2. Registry

```ts
// server/src/hooks/HooksRegistry.ts
import { logger } from '../utils/logger.js';
import type { HookEvent, HookHandler, HookOptions, HookContext } from './types.js';

interface RegisteredHook {
	handler:    HookHandler;
	priority:   number;
	collection?: string;
	blocking:   boolean;
}

const registry = new Map<HookEvent, RegisteredHook[]>();

export function registerHook(event: HookEvent, handler: HookHandler, opts: HookOptions = {}): void {
	const list = registry.get(event) ?? [];
	list.push({
		handler,
		priority:   opts.priority ?? 100,
		collection: opts.collection,
		blocking:   opts.blocking ?? false,
	});
	list.sort((a, b) => a.priority - b.priority);
	registry.set(event, list);
}

export function clearHooks(event?: HookEvent): void {
	if (event) registry.delete(event);
	else       registry.clear();
}

export async function dispatch(event: HookEvent, ctx: HookContext): Promise<void> {
	const list = registry.get(event);
	if (!list?.length) return;
	for (const h of list) {
		if (h.collection && h.collection !== ctx.collection) continue;
		try {
			await h.handler(ctx);
		} catch (err) {
			logger.error(`Hook ${event} failed (collection=${ctx.collection}):`, err);
			if (h.blocking) throw err;
		}
	}
}
```

## 3. Built-in hooks

```ts
// server/src/hooks/builtins.ts
import { registerHook } from './HooksRegistry.js';
import { logAudit } from '../services/AuditService.js';
import { broadcastToTable } from '../socket/index.js';
import { getDomainActions } from '../models/domainActions.js';

export function registerBuiltinHooks(): void {
	// AUDIT — priority 50, all events
	registerHook('post:create', (ctx) => {
		logAudit({
			action: 'create',
			userId: ctx.user?.userId,
			login:  ctx.user?.login,
			resourceType: ctx.collection,
			resourceId:   String((ctx.data as any)?._id ?? ctx.recordId),
			status: 'success',
			ipAddress: ctx.req?.ipAddress,
			userAgent: ctx.req?.userAgent,
		});
	}, { priority: 50 });

	registerHook('post:update', (ctx) => {
		logAudit({
			action: 'update',
			userId: ctx.user?.userId,
			login:  ctx.user?.login,
			resourceType: ctx.collection,
			resourceId:   ctx.recordId,
			status: 'success',
			details: { fields: Object.keys((ctx.data as any) ?? {}) },
			ipAddress: ctx.req?.ipAddress,
			userAgent: ctx.req?.userAgent,
		});
	}, { priority: 50 });

	registerHook('post:delete', (ctx) => {
		logAudit({
			action: 'delete',
			userId: ctx.user?.userId,
			login:  ctx.user?.login,
			resourceType: ctx.collection,
			resourceId:   ctx.recordId,
			status: 'success',
			details: ctx.data as Record<string, unknown>,
		});
	}, { priority: 50 });

	// BROADCAST — priority 80
	registerHook('post:create',  (ctx) => broadcastToTable(ctx.collection, 'data:created',  ctx.data));
	registerHook('post:update',  (ctx) => broadcastToTable(ctx.collection, 'data:updated',  { id: ctx.recordId, record: ctx.data }));
	registerHook('post:delete',  (ctx) => broadcastToTable(ctx.collection, 'data:deleted',  { id: ctx.recordId }));
	registerHook('post:restore', (ctx) => broadcastToTable(ctx.collection, 'data:restored', { id: ctx.recordId }));

	// DOMAIN ACTIONS — priority 90 (after audit/broadcast)
	registerHook('post:create', async (ctx) => {
		const da = getDomainActions(ctx.collection);
		if (da?.afterCreate) await da.afterCreate(ctx.data, ctx.collection);
	}, { priority: 90 });

	registerHook('post:update', async (ctx) => {
		const da = getDomainActions(ctx.collection);
		if (da?.afterUpdate) await da.afterUpdate(ctx.data, ctx.collection);
	}, { priority: 90 });

	registerHook('pre:delete', async (ctx) => {
		const da = getDomainActions(ctx.collection);
		if (da?.beforeDelete) await da.beforeDelete(ctx.recordId!, ctx.collection);
	}, { priority: 90, blocking: true });   // beforeDelete peut veto
}
```

## Acceptance criteria

1. `registerHook('post:create', fn)` → `fn` invoqué après chaque création
2. Plusieurs hooks même event → exécutés dans l'ordre `priority` (asc)
3. Hook non-blocking qui throw → log error + continue les suivants
4. Hook blocking qui throw → propage l'erreur, abort l'opération
5. `collection: 'x'` filtre → hook ne tire que pour cette collection
6. Audit/broadcast/domainActions = hooks built-in, plus de code inline dans `data.ts`
7. Tests existants `data.test.ts` + `domainActions.test.ts` passent sans modification
8. `permissions.test.ts`, `schemeValidation.test.ts` passent — aucune régression
