### 2. Extraire les règles de validation en module partagé

Si `MachineSchemeValidate` est couplé à Svelte/browser → extraire les règles pures dans un module TypeScript partagé :

```ts
// src/lib/main/machine/validateRules.ts  (OU server/src/validation/rules.ts)
// Règles pures, pas de dépendance Svelte/browser

export type FieldRule = {
  required?: boolean;
  type?: string;      // 'email'|'url'|'date'|'number'|'integer'|...
  min?: number;
  max?: number;
  pattern?: RegExp;
};

export function validateField(value: unknown, rule: FieldRule): string | null {
  if (rule.required && (value === undefined || value === null || value === ''))
    return 'Ce champ est obligatoire';
  if (value === undefined || value === null || value === '') return null;

  switch (rule.type) {
    case 'email':
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(String(value)))
        return 'Format email invalide';
      break;
    case 'url':
      if (!/^https?:\/\/.+/.test(String(value)))
        return 'URL doit commencer par http:// ou https://';
      break;
    case 'date':
      if (isNaN(Date.parse(String(value))))
        return 'Format de date invalide';
      break;
    case 'number':
      if (isNaN(Number(value)))
        return 'Doit être un nombre';
      break;
    case 'integer':
      if (!Number.isInteger(Number(value)))
        return 'Doit être un entier';
      break;
  }
  if (rule.min !== undefined && Number(value) < rule.min)
    return `Minimum ${rule.min}`;
  if (rule.max !== undefined && Number(value) > rule.max)
    return `Maximum ${rule.max}`;
  return null;
}

export function validateRecord(
  data: Record<string, unknown>,
  fields: Record<string, FieldRule>
): { valid: boolean; errors: Record<string, string> } {
  const errors: Record<string, string> = {};
  for (const [name, rule] of Object.entries(fields)) {
    const error = validateField(data[name], rule);
    if (error) errors[name] = error;
  }
  return { valid: Object.keys(errors).length === 0, errors };
}
```

### 3. Brancher dans createRecord / updateRecord

Problème : pour valider, on a besoin du `template.fields` de la collection.  
Options :
- **A** (simple) : le domain `actions.ts` fournit les règles via `validate()` — déjà dans S16-03
- **B** (complet) : fetcher le schéma depuis MongoDB (`appscheme_*`) et valider generiquement

> Assumed: commencer par Option A (validation via domain actions.validate()). Option B = S17+ après que le scheme cache server-side soit établi.

```ts
// Dans createRecord, avant Model.create():
const domainActions = domainActionsRegistry.get(table);
if (domainActions?.validate) {
  const { valid, errors } = domainActions.validate(req.body, table);
  if (!valid) {
    res.status(422).json({ error: 'Validation failed', errors });
    return;
  }
}
```
