## User Story

As a **QA engineer**, I want to **write 30+ test cases for the `MachineParserForge` DSL parser**, so that **field type detection (text, number, fk, array-of, modifiers) is verified against all expected syntax variations**.

## Context

The baseline audit found that `machineParserForge.ts` is the core DSL parser but has **ZERO test coverage**. This is a critical gap: if the parser breaks, all field type detection fails.

Expected test coverage:

- Primitive types: `text`, `number`, `boolean`, `date`, `email`
- Complex types: `fk-collection.keyField`, `array-of-type`
- Modifiers: `(required)`, `(readonly)`, `(private)`
- Edge cases: empty string, malformed DSL, unknown types

    Given DSL: "fk-teams.id (required)"

    Then returns IDbForge with private: true

  Scenario: Parse readonly field
    Given DSL: "id (readonly)"

## Test File Template

**Location**: `src/lib/main/__tests__/machineParserForge.test.ts` (new file)

```typescript
import { describe, it, expect } from 'vitest';
import { MachineParserForge } from '$lib/main/machineParserForge.js';

describe('MachineParserForge', () => {
	const parser = new MachineParserForge();

	describe('Primitive Types', () => {
		it('parses "text" field', () => {
			const result = parser.parse('text');
			expect(result.type).toBe('text');
			expect(result.required).toBe(false);
		});

		it('parses "text (required)" field', () => {
			const result = parser.parse('text (required)');
			expect(result.type).toBe('text');
			expect(result.required).toBe(true);
		});

		it('parses "number" field', () => {
			const result = parser.parse('number');
			expect(result.type).toBe('number');
		});

		it('parses "boolean" field', () => {
			const result = parser.parse('boolean');
			expect(result.type).toBe('boolean');
		});

		it('parses "date" field', () => {
			const result = parser.parse('date');
			expect(result.type).toBe('date');
		});

		it('parses "email" field', () => {
			const result = parser.parse('email');
			expect(result.type).toBe('email');
		});
	});

	describe('Foreign Keys', () => {
		it('parses "fk-teams.id" field', () => {
			const result = parser.parse('fk-teams.id');
			expect(result.type).toBe('fk');
			expect(result.format).toBe('teams.id');
		});

		it('parses "fk-teams.id (required)" field', () => {
			const result = parser.parse('fk-teams.id (required)');
			expect(result.type).toBe('fk');
			expect(result.required).toBe(true);
		});
	});

	describe('Array Types', () => {
		it('parses "array-of-string" field', () => {
			const result = parser.parse('array-of-string');
			expect(result.type).toBe('array');
			expect(result.itemType).toBe('string');
		});

		it('parses "array-of-number" field', () => {
			const result = parser.parse('array-of-number');
			expect(result.itemType).toBe('number');
		});
	});

	describe('Modifiers', () => {
		it('parses "(private)" modifier', () => {
			const result = parser.parse('text (private)');
			expect(result.private).toBe(true);
		});

		it('parses "(readonly)" modifier', () => {
			const result = parser.parse('id (readonly)');
			expect(result.readonly).toBe(true);
		});

		it('parses multiple modifiers', () => {
			const result = parser.parse('text (required private)');
			expect(result.required).toBe(true);
			expect(result.private).toBe(true);
		});

		it('parses all modifiers combined', () => {
			const result = parser.parse('text (required readonly private)');
			expect(result.required).toBe(true);
			expect(result.readonly).toBe(true);
			expect(result.private).toBe(true);
		});
	});

	describe('Edge Cases', () => {
		it('handles extra whitespace', () => {
			const result = parser.parse('text  (required)');
			expect(result.type).toBe('text');
			expect(result.required).toBe(true);
		});

		it('throws on malformed DSL', () => {
			expect(() => parser.parse('invalid-@#$')).toThrow();
		});

		it('handles empty string gracefully', () => {
			expect(() => parser.parse('')).toThrow();
		});
	});
});
```
