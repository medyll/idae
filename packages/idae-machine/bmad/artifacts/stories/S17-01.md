`server/src/validation/SchemeValidator.ts`

```ts
import { getConn } from '../middleware/dbRouter.js';
import { config } from '../config.js';
import type { FieldRule } from './validateRules.js';
import { validateRecord, type ValidationResult } from './validateRules.js';

// ── Type mapping ─────────────────────────────────────────────────────────────

const TYPE_MAP: Record<string, string> = {
  number:   'number',
  integer:  'integer',
  boolean:  'boolean',
  email:    'email',
  url:      'url',
  date:     'date',
  datetime: 'datetime',
  time:     'time',
  phone:    'phone',
  currency: 'currency',
  password: 'password',
};

// text and text-* variants → 'text' (no format constraint, only required/length)
const SKIP_TYPES = new Set(['id', 'image', 'file', 'json', 'any', 'fk']);

function mapType(appschemeType: string): string | undefined {
  if (SKIP_TYPES.has(appschemeType)) return undefined;
  if (appschemeType.startsWith('fk-')) return undefined;
  if (appschemeType.startsWith('text')) return 'text';
  return TYPE_MAP[appschemeType];
}

// ── Cache ─────────────────────────────────────────────────────────────────────

const schemeCache = new Map<string, Record<string, FieldRule>>();

export function invalidateSchemeCache(collection?: string): void {
  if (collection) schemeCache.delete(collection);
  else schemeCache.clear();
}

// ── Fetch from MongoDB ────────────────────────────────────────────────────────

async function fetchSchemeRules(collection: string): Promise<Record<string, FieldRule>> {
  const conn = await getConn(`${config.org}_machine_app`);

  // 1. Get has_field docs for this collection
  const hasDocs = await conn
    .collection('appscheme_has_field')
    .find({ 'gridFks.appscheme.code': collection })
    .toArray();

  if (hasDocs.length === 0) return {};

  // 2. Collect field names, fetch field type info
  const fieldNames = hasDocs.map((d: any) => d.gridFks?.appscheme_field?.code).filter(Boolean);

  const fieldDocs = await conn
    .collection('appscheme_field')
    .find({ code: { $in: fieldNames } })
    .toArray();

  const fieldTypeByCode = new Map<string, string>(
    fieldDocs.map((d: any) => [d.code as string, d.gridFks?.appscheme_field_type?.code ?? 'text'])
  );

  // 3. Build FieldRule map
  const rules: Record<string, FieldRule> = {};

  for (const doc of hasDocs) {
    const fieldCode = doc.gridFks?.appscheme_field?.code;
    if (!fieldCode) continue;

    const appType  = fieldTypeByCode.get(fieldCode) ?? 'text';
    const mapped   = mapType(appType);

    rules[fieldCode] = {
      required: doc.required === 1,
      ...(mapped ? { type: mapped } : {}),
    };
  }

  return rules;
}

// ── Public API ────────────────────────────────────────────────────────────────

/**
 * Get FieldRule map for a collection. Cached after first fetch.
 * Returns empty object if collection not in appscheme (graceful skip).
 */
export async function getSchemeRules(collection: string): Promise<Record<string, FieldRule>> {
  if (schemeCache.has(collection)) return schemeCache.get(collection)!;

  try {
    const rules = await fetchSchemeRules(collection);
    schemeCache.set(collection, rules);
    return rules;
  } catch (err) {
    // Schema not available — validation skipped, not a hard failure
    return {};
  }
}

/**
 * Validate a record against the appscheme rules for a collection.
 * Returns { valid: true } if collection has no schema (graceful skip).
 */
export async function validateAgainstScheme(
  collection: string,
  data: Record<string, unknown>
): Promise<ValidationResult> {
  const rules = await getSchemeRules(collection);
  if (Object.keys(rules).length === 0) return { valid: true, errors: {} };
  return validateRecord(data, rules);
}
```

## Acceptance criteria

- [ ] `getSchemeRules('vehicule')` → FieldRule map avec `required` correct pour chaque champ
- [ ] Cache hit sur 2e appel (pas de requête MongoDB)
- [ ] Collection absente → `{}` retourné, pas d'exception
- [ ] `mapType('fk-role.id')` → `undefined` (skippé)
- [ ] `mapType('text-area')` → `'text'`
- [ ] `mapType('email')` → `'email'`
- [ ] `invalidateSchemeCache()` vide tout, `invalidateSchemeCache('vehicule')` vide seulement vehicule
