```typescript
// ============================================================
// subscribe()
// ============================================================

describe('subscribe()', () => {
  it('calls listener on mutation', () => {
    const state = stator({ x: 0 });
    const fn = vi.fn();
    state.subscribe(fn);
    state.value.x = 1;
    expect(fn).toHaveBeenCalledTimes(1);
  });

  it('listener receives (oldValue, newValue)', () => {
    const state = stator({ x: 0 });
    let captured: any = null;
    state.subscribe((oldVal, newVal) => { captured = { oldVal, newVal }; });
    state.value.x = 42;
    // oldVal and newVal are the root state object
    expect(captured).not.toBeNull();
    // After mutation, newVal should reflect updated state
    expect(captured.newVal.x ?? (captured.newVal as any)?.x).toBeDefined();
  });

  it('unsubscribe stops notifications', () => {
    const state = stator({ x: 0 });
    const fn = vi.fn();
    const unsub = state.subscribe(fn);
    state.value.x = 1;
    unsub();
    state.value.x = 2;
    expect(fn).toHaveBeenCalledTimes(1); // only the first mutation
  });

  it('multiple subscribers are independent', () => {
    const state = stator({ x: 0 });
    const fn1 = vi.fn();
    const fn2 = vi.fn();
    const unsub1 = state.subscribe(fn1);
    state.subscribe(fn2);
    state.value.x = 1;
    unsub1();
    state.value.x = 2;
    expect(fn1).toHaveBeenCalledTimes(1); // stopped after unsub
    expect(fn2).toHaveBeenCalledTimes(2); // still receiving
  });

  it('unsubscribe is idempotent (calling twice does not throw)', () => {
    const state = stator({ x: 0 });
    const unsub = state.subscribe(vi.fn());
    expect(() => { unsub(); unsub(); }).not.toThrow();
  });
});

// ============================================================
// batch()
// ============================================================

describe('batch() — instance method', () => {
  it('groups multiple mutations into one notification', () => {
    const state = stator({ x: 0, y: 0 });
    const fn = vi.fn();
    state.subscribe(fn);
    state.batch(() => {
      state.value.x = 1;
      state.value.y = 2;
    });
    expect(fn).toHaveBeenCalledTimes(1);
  });

  it('fires no notification if nothing changed inside batch', () => {
    const state = stator({ x: 0 });
    const fn = vi.fn();
    state.subscribe(fn);
    state.batch(() => {
      // no mutation
    });
    expect(fn).toHaveBeenCalledTimes(0);
  });

  it('resumes immediate notifications after batch ends', () => {
    const state = stator({ x: 0 });
    const fn = vi.fn();
    state.subscribe(fn);
    state.batch(() => { state.value.x = 1; });
    state.value.x = 2; // outside batch — immediate
    expect(fn).toHaveBeenCalledTimes(2); // one from batch, one immediate
  });

  it('does not suppress notification if exception thrown inside batch', () => {
    const state = stator({ x: 0 });
    const fn = vi.fn();
    state.subscribe(fn);
    try {
      state.batch(() => {
        state.value.x = 1;
        throw new Error('boom');
      });
    } catch (_) { /* expected */ }
    // Pending notifications should still flush despite the error
    expect(fn).toHaveBeenCalledTimes(1);
  });
});

describe('batch() — standalone import', () => {
  it('batch is importable from the package', async () => {
    // Verify the standalone export works
    const { batch } = await import('$lib/stator/batch.js');
    expect(typeof batch).toBe('function');
  });

  it('standalone batch groups mutations across multiple stator instances', () => {
    const { batch } = require('$lib/stator/batch.js') ?? { batch: null };
    // Fallback if require not available: use the instance method test above
    if (!batch) return;

    const s1 = stator({ a: 0 });
    const s2 = stator({ b: 0 });
    const fn1 = vi.fn();
    const fn2 = vi.fn();
    s1.subscribe(fn1);
    s2.subscribe(fn2);
    batch(() => {
      s1.value.a = 1;
      s2.value.b = 1;
    });
    expect(fn1).toHaveBeenCalledTimes(1);
    expect(fn2).toHaveBeenCalledTimes(1);
  });
});
```
