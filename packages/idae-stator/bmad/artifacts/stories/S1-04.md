# S1-04 — Créer `proxy.ts` — isoler le moteur Proxy
**Effort:** M | **Sprint:** 1 | **Dépend de:** S1-02

## Contexte
Les fonctions `createHandler()` et `createProxy()` ainsi que la WeakMap `proxyCache` sont actuellement à l'intérieur de la fonction `stator()` dans `Stator.ts`. Il faut les extraire dans `proxy.ts`. Le `proxyCache` reste géré localement dans `stator()` mais est passé via `HandlerContext`.

## Fichier à CRÉER
`src/lib/stator/proxy.ts`

## Contenu EXACT du fichier `proxy.ts` à créer

```typescript
import type { HandlerContext, Primitive } from './types.js';

/**
 * Creates a ProxyHandler (get/set/deleteProperty/ownKeys/getOwnPropertyDescriptor/has)
 * for either the root StateWrapper or a nested object.
 *
 * @param isRoot - true for the root proxy (wraps StateWrapper<T>), false for nested objects
 * @param ctx    - shared context injected from stator()
 */
export function createHandler<T>(isRoot: boolean, ctx: HandlerContext<T>): ProxyHandler<any> {

  function getTarget(target: any): any {
    return isRoot ? target._value : target;
  }

  function getRawValue(target: any): any {
    return isRoot ? target._value : target;
  }

  function getReservedProperties(target: any): Record<string | symbol, any> {
    const rawValue = getRawValue(target);

    const reserved: Record<string | symbol, any> = {
      onchange: ctx.getOnChange(),
      addEventListener: ctx.eventTarget.addEventListener,
      removeEventListener: ctx.eventTarget.removeEventListener,
      triggerChange: ctx.eventTarget.triggerChange,
      valueOf: () => rawValue,
      toString: () => ctx.isPrimitive(rawValue) ? String(rawValue) : JSON.stringify(rawValue),
      [Symbol.for('nodejs.util.inspect.custom')]: () => rawValue,
    };

    if (isRoot) {
      reserved.value = getTarget(target);
      reserved.stator = getTarget(target);
    }

    return reserved;
  }

  return {
    get(target, property, receiver) {
      const reserved = getReservedProperties(target);

      if (property in reserved) {
        return reserved[property];
      }

      if (property === Symbol.toPrimitive) {
        const rawValue = getRawValue(target);
        return (hint: string) => {
          if (ctx.isPrimitive(rawValue)) {
            return hint === 'number' ? Number(rawValue) : String(rawValue);
          }
          return hint === 'number' ? Number(rawValue) : JSON.stringify(rawValue);
        };
      }

      const actualTarget = getTarget(target);

      if (ctx.isPrimitive(actualTarget)) {
        return isRoot ? Reflect.get(target, property, receiver) : undefined;
      }

      const value = Reflect.get(actualTarget, property, receiver);
      return ctx.isPrimitive(value) ? value : ctx.createDeepProxy(value);
    },

    set(target, property, value, receiver) {
      if (property === 'onchange') {
        if (typeof value === 'function') {
          ctx.setOnChange(value);
        } else if (value == null) {
          ctx.setOnChange(undefined);
        } else {
          throw new TypeError('onchange must be a function or undefined');
        }
        return true;
      }

      if (isRoot && (property === 'value' || property === 'stator')) {
        const oldValue = target._value;
        target._value = ctx.isPrimitive(value) ? value : ctx.createDeepProxy(value);
        ctx.notify(oldValue, target._value);
        return true;
      }

      const actualTarget = getTarget(target);

      if (ctx.isPrimitive(actualTarget)) {
        return false;
      }

      const oldValue = actualTarget[property];
      const proxiedValue = ctx.isPrimitive(value) ? value : ctx.createDeepProxy(value);
      const result = Reflect.set(actualTarget, property, proxiedValue, receiver);

      if (oldValue !== proxiedValue) {
        ctx.notify(ctx.rootState._value, ctx.rootState._value);
      }
      return result;
    },

    deleteProperty(target, property) {
      const actualTarget = getTarget(target);

      if (ctx.isPrimitive(actualTarget)) {
        return false;
      }

      const result = Reflect.deleteProperty(actualTarget, property);
      if (result) {
        ctx.notify(ctx.rootState._value, ctx.rootState._value);
      }
      return result;
    },

    ownKeys(target) {
      const actualTarget = getTarget(target);
      if (ctx.isPrimitive(actualTarget)) {
        return isRoot ? Reflect.ownKeys(target) : [];
      }
      return Reflect.ownKeys(actualTarget);
    },

    getOwnPropertyDescriptor(target, property) {
      const reservedKeys = ['onchange', 'addEventListener', 'removeEventListener', 'triggerChange'];
      if (reservedKeys.includes(property as string)) {
        return undefined;
      }

      const actualTarget = getTarget(target);
      if (ctx.isPrimitive(actualTarget)) {
        return isRoot ? Reflect.getOwnPropertyDescriptor(target, property) : undefined;
      }
      return Reflect.getOwnPropertyDescriptor(actualTarget, property);
    },

    has(target, property) {
      const reserved = getReservedProperties(target);
      if (property in reserved) return true;

      const actualTarget = getTarget(target);
      if (ctx.isPrimitive(actualTarget)) {
        return isRoot ? Reflect.has(target, property) : false;
      }
      return Reflect.has(actualTarget, property);
    },
  };
}

/**
 * Creates a Proxy for the given object using the provided context.
 * Uses proxyCache (from ctx) to avoid wrapping the same object twice.
 *
 * @param obj    - the object to wrap (or StateWrapper for root)
 * @param isRoot - true only for the root StateWrapper proxy
 * @param ctx    - shared context
 */
export function createProxy<T>(obj: any, isRoot: boolean, ctx: HandlerContext<T>): any {
  if (ctx.isPrimitive(obj)) return obj;
  if (!isRoot && ctx.proxyCache.has(obj)) return ctx.proxyCache.get(obj);

  const handler = createHandler(isRoot, ctx);
  const proxy = new Proxy(obj, handler);

  if (!isRoot) {
    ctx.proxyCache.set(obj, proxy);
  }

  return proxy;
}
```

## Fichier à MODIFIER : `src/lib/stator/Stator.ts`

### Étape 1 : Ajouter l'import
```typescript
import { createProxy } from './proxy.js';
```

### Étape 2 : Supprimer les fonctions `createHandler` et `createProxy` de `Stator.ts`
Ces deux fonctions (définies à l'intérieur de `stator()`) doivent être supprimées.

### Étape 3 : Adapter les usages dans `stator()`

Remplacer :
```typescript
createDeepProxy: (obj: any) => createProxy(obj, false),
```
Par :
```typescript
createDeepProxy: (obj: any) => createProxy(obj, false, ctx),
```

Remplacer les deux appels `createProxy(...)` dans `stator()` :
```typescript
// initialisation
stateWrapper._value = isPrimitive(initialState) ? initialState : createProxy(initialState, false);
// return
return createProxy(stateWrapper, true) as unknown as AugmentedState<T>;
```
Par :
```typescript
stateWrapper._value = isPrimitive(initialState) ? initialState : createProxy(initialState, false, ctx);
return createProxy(stateWrapper, true, ctx) as unknown as AugmentedState<T>;
```

**Attention :** `ctx` doit être défini avant d'être utilisé dans `createDeepProxy`. L'ordre dans `stator()` doit être :
1. Déclarer `stateWrapper`
2. Déclarer les fonctions helpers (`isPrimitive`, `notify`, `setOnChange`)
3. Construire `ctx` (en référençant `createDeepProxy` via `(obj) => createProxy(obj, false, ctx)`)
4. Initialiser `stateWrapper._value`
5. Retourner `createProxy(stateWrapper, true, ctx)`

## Pièges à éviter
- `ctx` est créé dans `stator()` et passé en paramètre — `proxy.ts` ne crée PAS de contexte lui-même
- La référence circulaire `ctx.createDeepProxy → createProxy → ctx` est normale et intentionnelle
- Ne pas oublier de passer `ctx` à chaque appel de `createProxy`
