# S1-03 — Créer `polyfill.ts` — isoler EventTarget

La classe `StatorEventTarget` est actuellement définie à l'intérieur de la fonction `stator()` dans `Stator.ts`. Il faut l'extraire dans un fichier dédié `polyfill.ts` et exporter une fonction `getEventTarget()`.

## Contenu EXACT du fichier `polyfill.ts` à créer

```typescript
/**
 * Internal EventTarget polyfill to ensure cross-environment compatibility.
 * Supports: Browser (native), Legacy DOM (document.createElement), Node.js (custom registry).
 */
class StatorEventTarget {
  private et: EventTarget;

  constructor() {
    if (typeof window !== 'undefined' && typeof window.EventTarget !== 'undefined') {
      // Browser: use native EventTarget
      this.et = new window.EventTarget();
    } else if (typeof document !== 'undefined' && typeof document.createElement === 'function') {
      // Legacy DOM: use invisible element as bridge
      this.et = document.createElement('span');
    } else {
      // Node.js fallback: custom lightweight registry
      const listeners: Record<string, Set<EventListenerOrEventListenerObject>> = {};
      this.et = {
        addEventListener(type: string, listener: EventListenerOrEventListenerObject) {
          if (!listeners[type]) listeners[type] = new Set();
          listeners[type].add(listener);
        },
        removeEventListener(type: string, listener: EventListenerOrEventListenerObject) {
          listeners[type]?.delete(listener);
        },
        dispatchEvent(event: Event) {
          const ls = listeners[event.type];
          if (!ls) return true;
          for (const l of Array.from(ls)) {
            if (typeof l === 'function') l.call(undefined, event);
            else if (l && typeof l.handleEvent === 'function') l.handleEvent(event);
          }
          return true;
        }
      } as unknown as EventTarget;
    }
  }

  addEventListener(...args: Parameters<EventTarget['addEventListener']>) {
    this.et.addEventListener(...args);
  }

  removeEventListener(...args: Parameters<EventTarget['removeEventListener']>) {
    this.et.removeEventListener(...args);
  }

  triggerChange(event: Event): boolean {
    return this.et.dispatchEvent(event);
  }
}

/**
 * Returns a new cross-environment EventTarget instance.
 */
export function getEventTarget(): StatorEventTarget {
  return new StatorEventTarget();
}

/**
 * Type of the object returned by getEventTarget()
 */
export type StatorEventTargetType = {
  addEventListener: (...args: Parameters<EventTarget['addEventListener']>) => void;
  removeEventListener: (...args: Parameters<EventTarget['removeEventListener']>) => void;
  triggerChange: (event: Event) => boolean;
};
```

## Fichier à MODIFIER : `src/lib/stator/Stator.ts`

### Étape 1 : Ajouter l'import en haut du fichier
```typescript
import { getEventTarget } from './polyfill.js';
```

### Étape 2 : Supprimer la classe `StatorEventTarget` de `Stator.ts`
La classe entière (du `class StatorEventTarget {` jusqu'au `}` fermant) doit être supprimée.

### Étape 3 : Remplacer l'instanciation
Trouver cette ligne dans `stator()` :
```typescript
const eventTarget = new StatorEventTarget();
```
La remplacer par :
```typescript
const eventTarget = getEventTarget();
```
